### 3.1 Masterз»„д»¶

```python
class MasterNode:
    """
    MasterиҠӮзӮ№иҒҢиҙЈпјҡ
    1. зӣ‘еҗ¬WorkerиҝһжҺҘ
    2. з»ҙжҠӨWorkerжұ 
    3. д»»еҠЎи°ғеәҰеҲҶй…Қ
    4. з»“жһңж”¶йӣҶиҒҡеҗҲ
    5. ж•…йҡңжЈҖжөӢжҒўеӨҚ
    """
    
    def __init__(self, host: str, port: int):
        self.host = host
        self.port = port
        self.workers: Dict[str, WorkerConnection] = {}
        self.task_queue: asyncio.Queue = asyncio.Queue()
        self.results: Dict[str, TaskResult] = {}
        
    async def start(self):
        """еҗҜеҠЁMasterжңҚеҠЎ"""
        server = await asyncio.start_server(
            self._handle_worker,
            self.host,
            self.port
        )
        
        # еҗҜеҠЁи°ғеәҰеҷЁ
        asyncio.create_task(self._scheduler())
        
        async with server:
            await server.serve_forever()
    
    async def _scheduler(self):
        """д»»еҠЎи°ғеәҰеҷЁ"""
        while True:
            task = await self.task_queue.get()
            
            # йҖүжӢ©еҗҲйҖӮзҡ„Worker
            worker = self._select_worker(task)
            
            if worker:
                await self._assign_task(worker, task)
            else:
                # жІЎжңүеҸҜз”ЁWorkerпјҢж”ҫеӣһйҳҹеҲ—
                await asyncio.sleep(1)
                await self.task_queue.put(task)
    
    def _select_worker(self, task: Task) -> Optional[Worker]:
        """WorkerйҖүжӢ©зӯ–з•Ҙ"""
        for worker in self.workers.values():
            if worker.status == "idle" and self._check_resources(worker, task):
                return worker
        return None
```

### 3.2 Workerз»„д»¶

```python
class WorkerNode:
    """
    WorkerиҠӮзӮ№иҒҢиҙЈпјҡ
    1. иҝһжҺҘMaster
    2. жҺҘж”¶жү§иЎҢд»»еҠЎ
    3. жү§иЎҢд»»еҠЎйҖ»иҫ‘
    4. иҝ”еӣһжү§иЎҢз»“жһң
    5. е®ҡжңҹеҝғи·іжұҮжҠҘ
    """
    
    def __init__(self, master_host: str, master_port: int):
        self.master_host = master_host
        self.master_port = master_port
        self.worker_id = str(uuid.uuid4())
        self.status = "idle"
        
    async def start(self):
        """еҗҜеҠЁWorker"""
        reader, writer = await asyncio.open_connection(
            self.master_host,
            self.master_port
        )
        
        # жіЁеҶҢеҲ°Master
        await self._register(reader, writer)
        
        # еҗҜеҠЁеҝғи·і
        asyncio.create_task(self._heartbeat(writer))
        
        # зӣ‘еҗ¬д»»еҠЎ
        await self._listen_tasks(reader, writer)
    
    async def _execute_task(self, task: Task) -> TaskResult:
        """жү§иЎҢд»»еҠЎ"""
        self.status = "busy"
        
        try:
            # е®һйҷ…жү§иЎҢд»»еҠЎ
            result = await self._run_agent_logic(task)
            return TaskResult(success=True, output=result)
        except Exception as e:
            return TaskResult(success=False, error=str(e))
        finally:
            self.status = "idle"
```

```python
class FaultToleranceManager:
    """е®№й”ҷз®ЎзҗҶеҷЁ"""

            for worker in self.workers.values():

                    await self._handle_worker_failure(worker)
            await asyncio.sleep(5)

        """жЈҖжөӢWorkerжҳҜеҗҰеӨұж•Ҳ"""
```

```python
class RetryPolicy:
    """йҮҚиҜ•зӯ–з•Ҙ"""
    
    def __init__(self, max_retries: int = 3, backoff_factor: float = 2.0):
        self.max_retries = max_retries
        self.backoff_factor = backoff_factor
    
    def get_retry_delay(self, attempt: int) -> float:
        """жҢҮж•°йҖҖйҒҝ"""
        return (self.backoff_factor ** attempt)
    
    def should_retry(self, error: Exception, attempt: int) -> bool:
        """еҲӨж–ӯжҳҜеҗҰйҮҚиҜ•"""
        if attempt >= self.max_retries:
            return False
        
        # жҹҗдәӣй”ҷиҜҜдёҚйҮҚиҜ•пјҲеҰӮиҜӯжі•й”ҷиҜҜпјү
        non_retryable = [SyntaxError, TypeError]
        if any(isinstance(error, e) for e in non_retryable):
            return False
        
        return True
```

```python
class ConnectionPool:
    """TCPиҝһжҺҘжұ """
    
    def __init__(self, max_connections: int = 100):
        self.max_connections = max_connections
        self.connections: Queue = Queue()
        self.active_connections: int = 0
    
    async def acquire(self) -> Connection:
        """иҺ·еҸ–иҝһжҺҘ"""
        if not self.connections.empty():
            return await self.connections.get()
        
        if self.active_connections < self.max_connections:
            conn = await self._create_connection()
            self.active_connections += 1

    async def release(self, conn: Connection):
        """йҮҠж”ҫиҝһжҺҘ"""
        await self.connections.put(conn)
```
